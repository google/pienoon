46 // FontManager manages font rendering with OpenGL utilizing freetype
47 // and harfbuzz as a glyph rendering and layout back end.
48 //
49 // It opens speficied OpenType/TrueType font and rasterize to OpenGL texture.
50 // An application can use the generated texture for a text rendering.
51 //
52 // The class is not threadsafe, it's expected to be only used from
53 // within OpenGL rendering thread.

59  // an OpenGL texture sizes.

61  ~FontManager();

102  // The user of the font_manager class is expected to use the class in 2
103  // passes, a layout pass and a render pass.
104  // During the layout pass, the user invokes GetBuffer() calls to fetch all
105  // required glyphs into the cache. Once in the cache, the user can retrieve
106  // the size of the strings to position them.
107  // Once the layout pass finishes, the user invokes StartRenderPass() to upload
108  // cache images to
109  // the atlas texture, and use the texture in the render pass.
110  // This design helps to minimize the frequency of the atlas texture upload.
111  //
112  // Indicate a start of new layout pass. Call the API each time the user starts
113  // a layout pass.
114  void StartLayoutPass();
115 
116  // Flush existing glyph cache contents and start new layout pass.
117  // Call this API while in a layout pass when the glyph cache is fragmented.
118  void FlushAndUpdate() { UpdatePass(true); }
119 
120  // Indicate a start of new render pass. Call the API each time the user starts
121  // a render pass.
122  void StartRenderPass() { UpdatePass(false); }
123 
124  // Getter of the font atlas texture.
125  Texture *GetAtlasTexture() { return atlas_texture_.get(); }

127  // The user can supply a size selector function to adjust glyph sizes when
128  // storing a glyph cache entry.
129  // By doing that, multiple strings with slightly different sizes can share the
130  // same glyph cache entry so that the user can reduce a total # of cached
131  // entries.
132  // The callback argument is a requested glyph size in pixels.
133  // The callback should return desired glyph size that should be used for the
134  // glyph sizes.
135  void SetSizeSelector(std::function<int32_t(const int32_t)> selector) {
136  size_selector_.swap(selector);
137  }

249 // Font texture class inherits Texture publicly.
250 // The class has additional properties for font metrics.
251 //
252 // For details of font metrics, refer http://support.microsoft.com/kb/32667
253 // In this class, ascender and descender values are retrieved from FreeType
254 // font property.
255 // And internal/external leading values are updated based on rendering glyph
256 // information. When rendering a string, the leading values tracks max (min for
257 // internal leading) value in the string.
258 class FontMetrics {
259  public:
260  // Constructor
261  FontMetrics()
262  : base_line_(0),
263  internal_leading_(0),
264  ascender_(0),
265  descender_(0),
266  external_leading_(0) {}
267 
268  // Constructor with initialization parameters.
269  FontMetrics(const int32_t base_line, const int32_t internal_leading,
270  const int32_t ascender, const int32_t descender,
271  const int32_t external_leading)
272  : base_line_(base_line),
273  internal_leading_(internal_leading),
274  ascender_(ascender),
275  descender_(descender),
276  external_leading_(external_leading) {
277  assert(internal_leading >= 0);
278  assert(ascender >= 0);
279  assert(descender <= 0);
280  assert(external_leading <= 0);
281  }
282  ~FontMetrics() {}
283 
284  // Setter/Getter of the baseline value.
285  int32_t base_line() const { return base_line_; }
286  void set_base_line(const int32_t base_line) { base_line_ = base_line; }
287 
288  // Setter/Getter of the internal leading parameter.
289  int32_t internal_leading() const { return internal_leading_; }
290  void set_internal_leading(const int32_t internal_leading) {
291  assert(internal_leading >= 0);
292  internal_leading_ = internal_leading;
293  }
294 
295  // Setter/Getter of the ascender value.
296  int32_t ascender() const { return ascender_; }
297  void set_ascender(const int32_t ascender) {
298  assert(ascender >= 0);
299  ascender_ = ascender;
300  }
301 
302  // Setter/Getter of the descender value.
303  int32_t descender() const { return descender_; }
304  void set_descender(const int32_t descender) {
305  assert(descender <= 0);
306  descender_ = descender;
307  }
308 
309  // Setter/Getter of the external leading value.
310  int32_t external_leading() const { return external_leading_; }
311  void set_external_leading(const int32_t external_leading) {
312  assert(external_leading <= 0);
313  external_leading_ = external_leading;
314  }
315 
316  // Returns total height of the glyph.
317  int32_t total() const {
318  return internal_leading_ + ascender_ - descender_ - external_leading_;
319  }
320 
321  private:
322  // Baseline: Baseline of the glpyhs.
323  // When rendering multiple glyphs in the same horizontal group,
324  // baselines need be aligned.
325  // Most of glyphs fit within the area of ascender + descender.
326  // However some glyphs may render in internal/external leading area.
327  // (e.g. , underlines)
328  int32_t base_line_;
329 
330  // Internal leading: Positive value that describes the space above the
331  // ascender.
332  int32_t internal_leading_;
333 
334  // Ascender: Positive value that describes the size of the ascender above
335  // the baseline.
336  int32_t ascender_;
337 
338  // Descender: Negative value that describes the size of the descender below
339  // the baseline.
340  int32_t descender_;
341 
342  // External leading : Negative value that describes the space below
343  // the descender.
344  int32_t external_leading_;
345 };

375 // Font buffer class
376 // Used with the texture atlas rendering.
377 class FontBuffer {
378  public:
379  // Constants
380  static const int32_t kIndiciesPerCodePoint = 6;
381  static const int32_t kVerticesPerCodePoint = 4;
382 
383  FontBuffer() : revision_(0) {}
384 
385  // Constructor with a buffer sizse.
386  FontBuffer(uint32_t size) : revision_(0) {
387  indices_.reserve(size * kIndiciesPerCodePoint);
388  vertices_.reserve(size * kVerticesPerCodePoint);
389  code_points_.reserve(size);
390  }
391  ~FontBuffer() {}
392 
393  // Setter/Getter of the metrics parameter of the font texture.
394  const FontMetrics &metrics() const { return metrics_; }
395  void set_metrics(const FontMetrics &metrics) { metrics_ = metrics; }
396 
397  // Getter of the indices array.
398  std::vector<uint16_t> *get_indices() { return &indices_; }
399  const std::vector<uint16_t> *get_indices() const { return &indices_; }
400 
401  // Getter of the vertices array.
402  std::vector<FontVertex> *get_vertices() { return &vertices_; }
403  const std::vector<FontVertex> *get_vertices() const { return &vertices_; }
404 
405  // Getter of the code points array.
406  std::vector<uint32_t> *get_code_points() { return &code_points_; }
407  const std::vector<uint32_t> *get_code_points() const { return &code_points_; }
408 
409  // Getter/Setter of the size of the string.
410  const vec2i &get_size() const { return size_; }
411  void set_size(const vec2i &size) { size_ = size; }
412 
413  // Getter/Setter of the glyph cache revision counter.
414  // Each time a contents of the glyph cache is updated, the revision of the
415  // cache is updated.
416  // If the cache revision and the buffer revision is different, the
417  // font_manager try to re-construct the buffer.
418  uint32_t get_revision() const { return revision_; }
419  void set_revision(const uint32_t revision) { revision_ = revision; }
420 
421  // Getter/Setter of the pass counter.
422  // In the render pass, the value is used if the user of the class needs to
423  // call StartRenderPass() call to upload the atlas texture.
424  int32_t get_pass() const { return pass_; }
425  void set_pass(const int32_t pass) { pass_ = pass; }
426 
427  // Add 4 vertices used for a glyph rendering to the vertex array.
428  void AddVertices(const vec2 &pos, const int32_t base_line, const float scale,
429  const GlyphCacheEntry &entry);
430 
431  // Update UV information of a glyph entry.
432  // uv vector should include top left corner of UV value as xy, and
433  // bottom right of UV value s wz component of the vector.
434  void UpdateUV(const int32_t index, const vec4 &uv);
435 
436  // Verify sizes of arrays used in the buffer are correct.
437  bool Verify() {
438  assert(vertices_.size() == code_points_.size() * kVerticesPerCodePoint);
439  assert(indices_.size() == code_points_.size() * kIndiciesPerCodePoint);
440  return true;
441  }
442 
443  private:
444  // Font metrics information.
445  FontMetrics metrics_;
446 
447  // Arrays for font indices, vertices and code points.
448  // They are hold as a separate vector because OpenGL draw call needs them to
449  // be a separate array.
450 
451  // Indices of the font buffer.
452  std::vector<uint16_t> indices_;
453 
454  // Vertices data of the font buffer.
455  std::vector<FontVertex> vertices_;
456 
457  // Code points used in the buffer. This array is used to fetch and update UV
458  // entries when the glyph cache is flushed.
459  std::vector<uint32_t> code_points_;
460 
461  // Size of the string in pixels.
462  vec2i size_;
463 
464  // Revision of the FontBuffer corresponding glyph cache revision.
465  // Caller needs to check the revision value if glyph texture has referencing
466  // entries by checking the revision.
467  uint32_t revision_;
468 
469  // Pass id. Each pass should have it's own texture atlas contents.
470  int32_t pass_;
471 };
