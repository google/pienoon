24 // ImpelProcessors that derive from ImpelProcessorWithVelocity should have an
25 // ImpelInit that derives from this struct (ImpelInitWithVelocity).
26 struct ImpelInitWithVelocity : public ImpelInit {
27  // A modular value wraps around from min to max. For example, an angle
28  // is modular, where -pi is equivalent to +pi. Setting this to true ensures
29  // that arithmetic wraps around instead of clamping to min/max.
30  bool modular;
31 
32  // Minimum value for Impeller::Value(). Clamped or wrapped-around when we
33  // reach this boundary.
34  float min;
35 
36  // Maximum value for Impeller::Value(). Clamped or wrapped-around when we
37  // reach this boundary.
38  float max;
39 
40  // Maximum speed at which the value can change. That is, maximum value for
41  // the Impeller::Velocity(). In units/tick.
42  // For example, if the value is an angle, then this is the max angular
43  // velocity, and the units are radians/tick.
44  float max_velocity;
45 
46  // Maximum that Impeller::Value() can be altered on a single call to
47  // ImpelEngine::AdvanceFrame(), regardless of velocity or delta_time.
48  float max_delta;
49 
50  // Cutoff to determine if the Impeller's current state has settled on the
51  // target. Once it has settled, Value() is set to TargetValue() and Velocity()
52  // is set to zero.
53  Settled1f at_target;
54 
55  // The derived type must call this constructor with it's ImpelType identifier.
56  explicit ImpelInitWithVelocity(ImpellerType type) :
57  ImpelInit(type),
58  modular(false),
59  min(0.0f),
60  max(0.0f),
61  max_velocity(0.0f),
62  max_delta(0.0f) {
63  }
64 
65  // Ensure position 'x' is within the valid constraint range.
66  float Normalize(float x) const {
67  // For non-modular values, do nothing.
68  if (!modular)
69  return x;
70 
71  // For modular values, ensures 'x' is in the constraints (min, max].
72  // That is, exclusive of min, inclusive of max.
73  const float width = max - min;
74  const float above_min = x <= min ? x + width : x;
75  const float normalized = above_min > max ? above_min - width : above_min;
76  assert(min < normalized && normalized <= max);
77  return normalized;
78  }
79 
80  // Ensure the Impeller's 'value' doesn't increment by more than 'max_delta'.
81  // This is different from ClampVelocity because it is independent of time.
82  // No matter how big the timestep, the delta will not be too great.
83  float ClampDelta(float delta) const {
84  return mathfu::Clamp(delta, -max_delta, max_delta);
85  }
86 
87  // Ensure velocity is within the reasonable limits.
88  float ClampVelocity(float velocity) const {
89  return mathfu::Clamp(velocity, -max_velocity, max_velocity);
90  }
91 
92  // Ensure the impeller value is within the specified range.
93  float ClampValue(float value) const {
94  return mathfu::Clamp(value, min, max);
95  }
96 
97  // Return true if we're close to the target and almost stopped.
98  // The definition of "close to" and "almost stopped" are given by the
99  // "at_target" member.
100  bool AtTarget(float dist, float velocity) const {
101  return at_target.Settled(dist, velocity);
102  }
103 };

106 // For impeller types that derive from ImpelInitWithVelocity, each Impeller
107 // gets a copy of this data. The data is held centrally, in the ImpelProcessor.
108 struct ImpelDataWithVelocity {
109  // What we are animating. Returned when Impeller::Value() called.
110  float value;
111 
112  // The rate of change of value. Returned when Impeller::Velocity() called.
113  float velocity;
114 
115  // What we are striving to hit. Returned when Impeller::TargetValue() called.
116  float target_value;
117 
118  virtual void Initialize(const ImpelInit& /*init_param*/) {
119  value = 0.0f;
120  velocity = 0.0f;
121  target_value = 0.0f;
122  }
123 };

126 // InitType must derive from ImpelInitWithVelocity.
127 // ImpelData must derive from ImpelDataWithVelocity. It should have a member
128 // called 'init' of type InitType.
129 template<class ImpelData, class InitType>
130 class ImpelProcessorWithVelocity : public ImpelProcessor<float> {
131  public:
132  virtual ~ImpelProcessorWithVelocity() {
133  assert(map_.Count() == 0);
134  }
135 
136  virtual ImpelId InitializeImpeller(const ImpelInit& init,
137  ImpelEngine* /*engine*/) {
138  assert(init.type == InitType::kType);
139 
140  // Allocate an external id, and map it to an index into data_.
141  ImpelId id = map_.Allocate();
142 
143  // Initialize the newly allocated item in data_.
144  Data(id).Initialize(static_cast<const InitType&>(init));
145  return id;
146  }
147 
148  virtual void RemoveImpeller(ImpelId id) { map_.Free(id); }
149  virtual ImpellerType Type() const { return InitType::kType; }
150 
151  // Accessors to allow the user to get and set simluation values.
152  virtual float Value(ImpelId id) const { return Data(id).value; }
153  virtual float Velocity(ImpelId id) const { return Data(id).velocity; }
154  virtual float TargetValue(ImpelId id) const { return Data(id).target_value; }
155  virtual void SetValue(ImpelId id, const float& value) {
156  Data(id).value = value;
157  }
158  virtual void SetVelocity(ImpelId id, const float& velocity) {
159  Data(id).velocity = velocity;
160  }
161  virtual void SetTargetValue(ImpelId id, const float& target_value) {
162  Data(id).target_value = target_value;
163  }
164  virtual void SetTargetTime(ImpelId /*id*/, float /*target_time*/) {}
165  virtual float Difference(ImpelId id) const {
166  const ImpelData& d = Data(id);
167  return d.init.Normalize(d.target_value - d.value);
168  }
169 
170  protected:
171  ImpelData& Data(ImpelId id) { return map_.Data(id); }
172  const ImpelData& Data(ImpelId id) const { return map_.Data(id); }
173 
174  IdMap<ImpelData> map_;
175 };
