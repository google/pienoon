31 static const float kPi = static_cast<float>(M_PI);
32 static const float kTwoPi = static_cast<float>(2.0 * M_PI);
33 static const float kThreePi = static_cast<float>(3.0 * M_PI);
34 static const float kHalfPi = static_cast<float>(M_PI_2);
35 static const float kDegreesToRadians = static_cast<float>(M_PI / 180.0);
36 static const float kRadiansToDegrees = static_cast<float>(180.0 / M_PI);
37 static const float kMaxUniqueAngle = kPi;
38 static const float kDegreesPerCircle = 360.0f;

40 // The biggest floating point number > -pi.
41 // That is, there are no floats x s.t. -pi < x < kMinUniqueAngle.
42 //
43 // Useful for clamping to the valid angle range:
44 // [kMinUniqueAngle,kMaxUniqueAngle] == (-pi, pi]
45 // Note: "[" indicates an inclusive bound; "(" indicates exclusive.
46 //
47 // In general, after modding, we should clamp to the valid range.
48 // Floating point precision errors can generate numbers outside of
49 // (-pi, pi], even when the math is perfect.
50 static const float kMinUniqueAngle = -3.1415925f;

53 // Purpose
54 // =======
55 // Represents an angle in radians, uniquely in the range (-pi, pi].
56 //
57 // We include pi in the range, but exclude -pi, because pi and -pi
58 // are equivalent mod 2pi.
59 //
60 // Equivalence is key to this class. We want only one representation
61 // of every equivalent angle. For example, 0 and 2pi are both represented
62 // as 0, internally. This unique representation allows for comparison and
63 // precise arithmetic.
64 //
65 // All operators keep the angle values in the valid range.
66 //
67 // Conversions to and from vectors and matrices
68 // ============================================
69 // Conversions from and to XZ vectors follow the following convension:
70 // The vector is a point on unit circle corresponding to a sweep of 'angle'
71 // degrees from the x-axis towards the z-axis.
72 // 0 ==> ( 1, 0)
73 // pi/2 ==> ( 0, 1)
74 // pi ==> (-1, 0)
75 // 3pi/2 ==> ( 0, -1)
76 //
77 // Why use instead of Quaternions?
78 // ===============================
79 // Quaternions are great for three dimensional rotations, but for many
80 // applications you only have two dimensional rotations. Instead of the
81 // four floats and heavy operations required by quaternions, an angle
82 // can represent the rotation in one float and reasonably light operations.
83 // Angles are easier to do trigonometry on. Also, they're conceptually
84 // simpler.
85 //
86 class Angle {
87  public:
88  Angle() : angle_(0.0f) {}

90  // Create from 'angle', which is already in the valid range (-pi,pi].
91  // If your angle is outside that range, construct the Angle with the

158  // Wraps an angle to the range (-pi, pi].
159  static float WrapAngle(float angle) {
160  angle -= (floor(angle / kTwoPi) + 1.f) * kTwoPi;
161  if (angle <= -kPi) {
162  angle += kTwoPi;
163  }
164  return angle;
165  }
166 
167  // Create from 'angle', in radians, which is in the range (-3pi,3pi].
168  // This function is significantly faster than FromRadians since it avoids
169  // division. It's also more precise for the same reason. The range may seem
170  // strange at first glance; it's a consequence of the implementation. Just
171  // know that any two sums of normalized angles will still be in the range
172  // (-3pi,3pi].
173  static Angle FromWithinThreePi(const float angle) {
174  return Angle(ModWithinThreePi(angle));
175  }
176 
177  // Create from 'radians', which is converted to the range (-pi, pi].
178  static Angle FromRadians(const float radians) {
179  return Angle(WrapAngle(radians));
180  }
181 
182  // Create from 'degrees', which is converted to the range (-pi, pi].
183  static Angle FromDegrees(const float degrees) {
184  return FromRadians(degrees * kDegreesToRadians);
185  }

187  // Create from the x,z coordinates of a vector, as described in the
188  // comment at the head of the class.
189  static Angle FromXZVector(const mathfu::vec3& v) {
190  return Angle(ModIfNegativePi(atan2f(v[2], v[0])));
191  }
192 
193  // Return true if 'angle' is within the valid range (-pi,pi], that is,
194  // the range inclusive of +pi but exclusive of -pi.
195  static bool IsAngleInRange(const float angle) {
196  return kMinUniqueAngle <= angle && angle <= kMaxUniqueAngle;
197  }

199  private:

201  friend bool operator<(const Angle& a, const Angle& b);
202  friend bool operator<=(const Angle& a, const Angle& b);

209  void ToVector(float* const x, float* const z) const {
210  *x = cos(angle_);
211  *z = sin(angle_);
212  }

214  // Take 'angle' in the range (-3pi,3pi] and return an equivalent angle in
215  // the range (-pi,pi]. Note that angles are equivalent if they differ by 2pi.
216  static float ModWithinThreePi(const float angle) {
217  assert(-kThreePi < angle && angle < kThreePi);
218  // These ternary operators should be converted into select statements by
219  // the compiler.
220  const float above = angle < kMinUniqueAngle ? angle + kTwoPi : angle;
221  const float below = above > kMaxUniqueAngle ? above - kTwoPi : above;
222  assert(IsAngleInRange(below));
223  return below;
224  }
225 
226  static float ModIfNegativePi(const float angle) {
227  // Pi negates to -pi, which is outside the range so becomes +pi again.
228  return angle < kMinUniqueAngle ? kMaxUniqueAngle : angle;
229  }

278 inline Angle Angle::Clamp(const Angle& center, const Angle& max_diff) const {
279  assert(0 <= max_diff.angle_ && max_diff.angle_ <= kPi);
280 
281  // Get difference from 'center'. We know this will be a value in the range
282  // (-pi, pi].
283  const Angle diff = (*this) - center;
284 
285  // Clamp the difference to the valid range.
286  const Angle diff_clamped(mathfu::Clamp(diff.angle_, -max_diff.angle_,
287  max_diff.angle_));
288 
289  // Add the difference onto the center. Note that, if no clamping happened,
290  // we're left with *this.
291  return center + diff_clamped;
292 }
