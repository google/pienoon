32 // Represents a shader consisting of a vertex and pixel shader. Also stores
33 // ids of standard uniforms. Use the Renderer class below to create these.
34 class Shader {
35  public:
36  Shader(GLuint program, GLuint vs, GLuint ps)
37  : program_(program),
38  vs_(vs),
39  ps_(ps),
40  uniform_model_view_projection_(-1),
41  uniform_model_(-1),
42  uniform_color_(-1),
43  uniform_light_pos_(-1),
44  uniform_camera_pos_(-1) {}
45 
46  ~Shader() {
47  if (vs_) GL_CALL(glDeleteShader(vs_));
48  if (ps_) GL_CALL(glDeleteShader(ps_));
49  if (program_) GL_CALL(glDeleteProgram(program_));
50  }
51 
52  // Will make this shader active for any subsequent draw calls, and sets
53  // all standard uniforms (e.g. mvp matrix) based on current values in
54  // Renderer, if this shader refers to them.
55  void Set(const Renderer &renderer) const;
56 
57  // Find a non-standard uniform by name, -1 means not found.
58  GLint FindUniform(const char *uniform_name) {
59  GL_CALL(glUseProgram(program_));
60  return glGetUniformLocation(program_, uniform_name);
61  }
62 
63 #ifdef _MSC_VER
64 #pragma warning(push)
65 #pragma warning(disable : 4127) // conditional expression is constant
66 #endif // _MSC_VER
67  // Set an non-standard uniform to a vec2/3/4 value.
68  // Call this after Set() or FindUniform().
69  template <int N>
70  void SetUniform(GLint uniform_loc, const mathfu::Vector<float, N> &value) {
71  // This should amount to a compile-time if-then.
72  if (N == 2) {
73  GL_CALL(glUniform2fv(uniform_loc, 1, &value[0]));
74  } else if (N == 3) {
75  GL_CALL(glUniform3fv(uniform_loc, 1, &value[0]));
76  } else if (N == 4) {
77  GL_CALL(glUniform4fv(uniform_loc, 1, &value[0]));
78  } else {
79  assert(0);
80  }
81  }
82 #ifdef _MSC_VER
83 #pragma warning(pop)
84 #endif // _MSC_VER
85 
86  // Convenience call that does a Lookup and a Set if found.
87  // Call this after Set().
88  template <int N>
89  bool SetUniform(const char *uniform_name,
90  const mathfu::Vector<float, N> &value) {
91  auto loc = FindUniform(uniform_name);
92  if (loc < 0) return false;
93  SetUniform(loc, value);
94  return true;
95  }
96 
97  bool SetUniform(const char *uniform_name, const float &value) {
98  auto loc = FindUniform(uniform_name);
99  if (loc < 0) return false;
100  GL_CALL(glUniform1f(loc, value));
101  return true;
102  }
103

107  GLuint program_, vs_, ps_;
108 
109  GLint uniform_model_view_projection_;
110  GLint uniform_model_;
111  GLint uniform_color_;
112  GLint uniform_light_pos_;
113  GLint uniform_camera_pos_;
114 };
