24 // Non-templatized base class of Impeller. Holds all the functions unrelated to
25 // the value being impelled. See comments above Impeller for details on
26 // Impellers.
27 class ImpellerBase {
28  public:
29  ImpellerBase() : processor_(nullptr), id_(kImpelIdInvalid) {}
30  ImpellerBase(const ImpelInit& init, ImpelEngine* engine) {
31  Initialize(init, engine);
32  }
33  ~ImpellerBase() { Invalidate(); }
34 
35  // Initialize this Impeller to the type specified in init.type.
36  void Initialize(const ImpelInit& init, ImpelEngine* engine) {
37  // Unregister ourselves with our existing ImpelProcessor.
38  Invalidate();
39 
40  // The ImpelProcessors are held centrally in the ImpelEngine. There is only
41  // one processor per type. Get that processor.
42  processor_ = engine->Processor(init.type);
43 
44  // Register and initialize ourselves with the ImpelProcessor.
45  id_ = processor_->InitializeImpeller(init, engine);
46  }
47 
48  // Detatch this Impeller from its ImpelProcessor. Functions other than
49  // Initialize can no longer be called after Invalidate has been called.
50  void Invalidate() {
51  if (Valid()) {
52  processor_->RemoveImpeller(id_);
53  processor_ = nullptr;
54  id_ = kImpelIdInvalid;
55  }
56  }
57 
58  // Return true if this Impeller is currently being driven by an
59  // ImpelProcessor. That is, it has been successfully initialized.
60  bool Valid() const { return processor_ != nullptr; }
61 
62  // Return the GUID representing the type Impeller we've been initilized to.
63  // An Impeller can take on any type, provided the dimensions match.
64  ImpellerType Type() const { return processor_->Type(); }
65 
66  // The number of floats (or doubles) that this Impeller is driving.
67  // For example, if this Impeller is driving a position in 3D space, then
68  // we will return 3 here. The dimension of an ImpellerBase is not fixed, but
69  // once we apply a templatization in Impeller<T>, then the dimension becomes
70  // fixed.
71  int Dimensions() const { return processor_->Dimensions(); }
72 
73  // Set simulation values. Some simulation values are independent of the
74  // number of dimensions, so we put them in the base class.
75  void SetTargetTime(float target_time) {
76  return processor_->SetTargetTime(id_, target_time);
77  }
78 
79  // More complicated impellers are aggregates of impellers. For example, a
80  // composite impeller might emit a 3D vector by aggregating three 1D
81  // impellers. These functions allow you to traverse the tree of impellers.
82  int ChildImpellerCount() const {
83  return processor_->ChildImpellerCount(id_);
84  }
85  const ImpellerBase* ChildImpeller(int i) const {
86  return processor_->ChildImpeller(id_, i);
87  }
88  ImpellerBase* ChildImpeller(int i) {
89  return processor_->ChildImpeller(id_, i);
90  }
91 
92  protected:
93  // All calls to an Impeller are proxied to an ImpellerProcessor. Impeller
94  // data and processing is centralized to allow for scalable optimizations
95  // (e.g. SIMD or parallelization).
96  ImpelProcessorBase* processor_;
97 
98  // An ImpelProcessor processes one ImpellerType, and hosts every Impeller of
99  // that type. The id here uniquely identifies this Impeller to the
100  // ImpelProcessor.
101  ImpelId id_;
102 };

104 // Impeller
105 // ========
106 // An Impeller drives a value of type Impeller::T. The value is driven towards
107 // a target that is set with Impeller::SetTargetValue(). The way the value moves
108 // towards the target is determined by the type of the Impeller, and its
109 // parameterization, which is set in Impeller::Initialize().
110 //
111 // The Impeller's current value can be queried via Impeller::Value().
112 // Additionally, some impeller types may also return Impeller::Velocity(), the
113 // derivative of value. Not every impeller type implements all functions,
114 // however. For example, the impeller type that drives 4x4 matrix
115 // transformations doesn't return a value for Velocity() because, for 4x4
116 // matrices the derivitive is not well defined.
117 //
118 //
119 // Converter
120 // =========
121 // The processors run on mathfu types, so the Converter is used to return
122 // values in your native math type.
123 //
124 // If you are using mathfu as your native math type, you can use the
125 // Impeller1f, Impeller2f, types defined below, and don't need to worry about
126 // Converter.
127 //
128 // If you have your own native math type that is byte-wise compatible with
129 // the mathfu types, you might use a union to convert between the two.
130 // Something like this:
131 // template<class MathFuT, class NativeT>
132 // class UnionConverter {
133 // union Union {
134 // MathFuT mathfu;
135 // NativeT native;
136 // };
137 // public:
138 // typedef NativeT ExternalApiType;
139 // static NativeT To(const MathFuT& x) {
140 // Union u;
141 // u.mathfu = x;
142 // return u.native;
143 // }
144 // static MathFuT From(const NativeT& x) {
145 // Union u;
146 // u.native = x;
147 // return u.mathfu;
148 // }
149 // };
150 // Note that you should be careful when checking that your types are byte-wise
151 // compatible with mathfu types. If MATHFU_COMPILE_WITH_PADDING is defined,
152 // three-dimensional vectors will have a 4th component that must be set
153 // properly. Unless you're sure, you should use constructors provided by
154 // mathfu and your native types.
155 //
156 template<class Converter>
157 class Impeller : public ImpellerBase {
158  public:
159  typedef typename Converter::ExternalApiType T;
160 
161  // Get current impeller values from the processor.
162  T Value() const { return Converter::To(Processor().Value(id_)); }
163  T Velocity() const { return Converter::To(Processor().Velocity(id_)); }
164  T TargetValue() const {
165  return Converter::To(Processor().TargetValue(id_));
166  }
167 
168  // Set current impeller values in the processor. Processors may choose to
169  // ignore whichever values make sense for them to ignore.
170  void SetValue(const T& value) {
171  return Processor().SetValue(id_, Converter::From(value));
172  }
173  void SetVelocity(const T& velocity) {
174  return Processor().SetVelocity(id_, Converter::From(velocity));
175  }
176  void SetTargetValue(const T& target_value) {
177  return Processor().SetTargetValue(id_, Converter::From(target_value));
178  }
179 
180  // Return the TargetValue() minus the Value(). If we're impelling a modular
181  // type (e.g. an angle), this may not be the naive subtraction.
182  T Difference() const { return Converter::To(Processor().Difference(id_)); }
183 
184  private:
185  typedef ImpelProcessor<T> ProcessorType;
186 
187  ProcessorType& Processor() {
188  return *static_cast<ProcessorType*>(processor_);
189  }
190 
191  const ProcessorType& Processor() const {
192  return *static_cast<const ProcessorType*>(processor_);
193  }
194 };

197 // The processors run on mathfu types, so these Converters can be used to return
198 // values in your native math type. This degenerate converter just sticks with
199 // the original mathfu type. Define your own if you're not using mathfu.
200 template<class T>
201 class MathFuConverter {
202  public:
203  typedef T ExternalApiType;
204  static T To(const T& x) { return x; }
205  static T From(const T& x) { return x; }
206 };
207 
208 // Define impellers that use mathfu types in the API. If you have your own
209 // vector library, you can define your own types with a suitable Converter.
210 // Note that ImpelProcessors use mathfu types, so the Converter is required to
211 // convert to and from your vector library to mathfu.
212 typedef Impeller< MathFuConverter<float> > Impeller1f;
213 typedef Impeller< MathFuConverter<mathfu::vec2> > Impeller2f;
214 typedef Impeller< MathFuConverter<mathfu::vec3> > Impeller3f;
215 typedef Impeller< MathFuConverter<mathfu::vec4> > Impeller4f;
216 typedef Impeller< MathFuConverter<mathfu::mat4> > ImpellerMatrix4f;
