22 // Any resources that can be loaded async should inherit from this.
23 class AsyncResource {
24  public:
25  AsyncResource(const std::string &filename)
26  : filename_(filename), data_(nullptr) {}
27  virtual ~AsyncResource() {}
28 
29  // Load should perform the actual loading of filename_, and store the
30  // result in data_, or nullptr upon failure. It is called on the loader
31  // thread, so should not access any program state outside of this object.
32  // Since there will be only one loader thread, any libraries called by Load
33  // need not be MT-safe as long as they're not also called by the main thread.
34  virtual void Load() = 0;
35 
36  // This should implement the behavior of turning data_ into the actual
37  // desired resource. Called on the main thread only.
38  virtual void Finalize() = 0;
39 
40  const std::string &filename() const { return filename_; }
41 
42  protected:
43  std::string filename_;
44  uint8_t *data_;
45 
46  friend class AsyncLoader;
47 };

49 class AsyncLoader {
50  public:
51  AsyncLoader();
52  ~AsyncLoader();
53 
54  // Call this any number of times before StartLoading.
55  void QueueJob(AsyncResource *res);
56 
57  // Launches the loading thread.
58  void StartLoading();
59 
60  // Cleans-up the background loading thread once all jobs have been completed.
61  // You can restart with StartLoading() if you like.
62  void StopLoadingWhenComplete();
63 
64  // Call this once per frame after StartLoading. Will call Finalize on any
65  // resources that have finished loading. One it returns true, that means
66  // the queue is empty, all resources have been processed, and the loading
67  // thread has terminated.
68  bool TryFinalize();
69 
70  private:
71  void Lock(const std::function<void()> &body) {
72  auto err = SDL_LockMutex(mutex_);
73  (void)err;
74  assert(err == 0);
75  body();
76  SDL_UnlockMutex(mutex_);
77  }
78  template <typename T>
79  T LockReturn(const std::function<T()> &body) {
80  T ret;
81  Lock([&ret, &body]() { ret = body(); });
82  return ret;
83  }
84 
85  void LoaderWorker();
86  static int LoaderThread(void *user_data);
87 
88  std::vector<AsyncResource *> queue_, done_;
89 
90  // Keep handle to the worker thread around so that we can wait for it to
91  // finish before destroying the class.
92  SDL_Thread *worker_thread_;
93 
94  // This lock protects ALL state in this class, i.e. the two vectors.
95  SDL_mutex *mutex_;
96 
97  // Kick-off the worker thread when a new job arrives.
98  SDL_semaphore *job_semaphore_;
99 };
